truvem_demo.py 
from truvem import Truvem

# One-time setup
client = Truvem(api_key="your_key")

# Save a memory
client.remember(
  agent_id="my-agent",
  content="User prefers dark mode"
)

# Recall — any session, any model
memories = client.recall("my-agent")
print(memories)
# {"status":"ok","memories":[...]}
